# UpstageDocumentParseLoader
- 문서의 구조를 분석해서 제목, 단락, 표, 이미지 등 다양한 요소를 자동으로 인식하고 추출이 가능함
- OCR도 제공해서 이미지로 저장된 PDF 문서에도 텍스트, 레이아웃 정보를 정밀하게 추출이 가능함

```python
from langchain_upstage import UpstageDocumentParseLoader

file_path = (
    "/Users/imkdw/study/RAG 비법노트/9. 문서 로더/maplestory-black-magition-pattern.pdf"
)

loader1 = UpstageDocumentParseLoader(
    file_path,
    output_format="html",
    split="element",
)

# page_content='<h1 id='0' style='font-size:18px'>나무위키 최근 변경 최근 토론 특수 기능</h1>' metadata={'id': 0, 'page': 1, 'category': 'heading1', 'coordinates': [{'x': 0.0203, 'y': 0.0082}, {'x': 0.6494, 'y': 0.0082}, {'x': 0.6494, 'y': 0.0327}, {'x': 0.0203, 'y': 0.0327}]}
# page_content='<p id='1' data-category='paragraph' style='font-size:18px'>여기에서 검색</p>' metadata={'id': 1, 'page': 1, 'category': 'paragraph', 'coordinates': [{'x': 0.0599, 'y': 0.0514}, {'x': 0.1656, 'y': 0.0514}, {'x': 0.1656, 'y': 0.0678}, {'x': 0.0599, 'y': 0.0678}]}
# page_content='<br><h1 id='2' style='font-size:14px'>→</h1>' metadata={'id': 2, 'page': 1, 'category': 'heading1', 'coordinates': [{'x': 0.9285, 'y': 0.05}, {'x': 0.9913, 'y': 0.05}, {'x': 0.9913, 'y': 0.069}, {'x': 0.9285, 'y': 0.069}]}
docs = loader1.load()
```
