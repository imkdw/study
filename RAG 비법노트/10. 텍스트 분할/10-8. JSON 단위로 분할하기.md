# JSON 단위로 분할하기
- `JsonRecursiveSplitter`는 JSON 데이터를 DFS 방식으로 탐색해서 더 작은 JSON 청크를 만듦
- JSON 데이터가 중간에 잘리지 않도록 중첩된 JSON 객체를 가능하면 유지할려고 하지만 청크 크기를 지정된 값의 사이를 맞추기 위해서는 분할함
- 값이 중첩된 JSON이 아닌 매우 큰 문자열인 경우는 해당 문자열은 분할되지 않음

```python
import requests
from langchain_text_splitters import RecursiveJsonSplitter

json_data = requests.get("https://api.smith.langchain.com/openapi.json").json()


splitter = RecursiveJsonSplitter(max_chunk_size=300)

# JSON 데이터를 재귀적으로 분할
json_chunks = splitter.split_json(json_data=json_data)

# JSON 데이터를 기반으로 문서를 생성
docs = splitter.create_documents(texts=[json_data])

# JSON 데이터를 기반으로 문자열 청크를 생성
texts = splitter.split_text(json_data=json_data)

# {
#     "openapi": "3.1.0",
#     "info": {"title": "LangSmith", "version": "0.1.0"},
#     "paths": {
#         "/api/v1/sessions/{session_id}/dashboard": {
#             "post": {
#                 "tags": ["tracer-sessions"],
#                 "summary": "Get Tracing Project Prebuilt Dashboard",
#                 "description": "Get a prebuilt dashboard for a tracing project.",
#             }
#         }
#     },
# }
print(docs[0].page_content)


# 분할된 문자열 청크를 출력합니다.
# {
#     "openapi": "3.1.0",
#     "info": {"title": "LangSmith", "version": "0.1.0"},
#     "paths": {
#         "/api/v1/sessions/{session_id}/dashboard": {
#             "post": {
#                 "tags": ["tracer-sessions"],
#                 "summary": "Get Tracing Project Prebuilt Dashboard",
#                 "description": "Get a prebuilt dashboard for a tracing project.",
#             }
#         }
#     },
# }
print(texts[0])
```
