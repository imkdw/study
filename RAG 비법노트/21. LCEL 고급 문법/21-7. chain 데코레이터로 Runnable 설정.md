# @chain 데코레이터로 Runnable 설정
```python
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import ChatPromptTemplate
from langchain_core.runnables import chain
from langchain_openai import ChatOpenAI

prompt1 = ChatPromptTemplate.from_template("{topic} 에 대해 짧게 한글로 설명해주세요.")

prompt2 = ChatPromptTemplate.from_template(
    "{sentence} 를 emoji를 활용한 인스타그램 게시글로 만들어주세요."
)


@chain
def custom_chain(text):
    # 먼저 요청된 문구에 대해서 짧게 한글로 작성함
    chain1 = prompt1 | ChatOpenAI(model="gpt-4o-mini") | StrOutputParser()
    output1 = chain1.invoke({"topic": text})

    chain2 = prompt2 | ChatOpenAI(model="gpt-4o-mini") | StrOutputParser()

    # 한글로 짧게 작성된 문장에 대해서 이모지를 사용한 인스타그램 게시글로 작성함
    return chain2.invoke({"sentence": output1})


# 🌌✨ 양자역학이란? ✨🌌

# 양자역학은 물리학의 한 분야로, 원자와 소립자와 같은 미시 세계의 물질과 에너지의 행동을 설명하는 신비로운 이론이에요! 🪐🔍

# 고전역학과는 달리, 여기서는 입자의 위치나 운동량이 정확히 결정되지 않고, 확률적으로 표현된답니다! 🎲🔮

# 입자는 파동과 입자의 성질을 모두 가지고 있어서, 양자 얽힘, 해밀토니안, 불확정성 원리와 같은 흥미로운 개념이 등장해요! 🤯🔗

# 양자역학은 현대 물리학의 기초를 형성하며, 반도체, 레이저, MRI 등 다양한 기술에 응용되고 있어요! 💡🔬

# #양자역학 #물리학 #미시세계 #확률 #과학 #기술 #현대물리학 #신비로운세계 🌌🔭✨
print(custom_chain.invoke("양자역학"))
```
